export default {
  async fetch(
    request: Request,
    env: Env,
    ctx: ExecutionContext
  ): Promise<Response> {
    return new Response(JSON.stringify(ctx), {
      headers: {
        'content-type': 'application/json;charset=UTF-8',
      },
    })
  },
}
